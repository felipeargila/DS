        const colunasVisitas = [

            'Visita Técnica',

            'PPO',

            'Enviado',

            'Negócio Gerado'

        ];

        

        let cotacoes = [];

        let visitas = [];

        

        // Inicialização

        document.addEventListener('DOMContentLoaded', function() {

            carregarDados();

            renderizarKanbans();

        });

        

        function carregarDados() {

            // Carregar do localStorage (depois substituiremos por API)

            cotacoes = JSON.parse(localStorage.getItem('cotacoes') || '[]');

            visitas = JSON.parse(localStorage.getItem('visitas') || '[]');

        }

        

        function renderizarKanbans() {

            renderizarKanbanCotacoes();

            renderizarKanbanVisitas();

            adicionarDragDrop();

        }

        

        function renderizarKanbanCotacoes() {

            const container = document.getElementById('kanbanCotacoes');

            container.innerHTML = '';

            

            colunasCotacoes.forEach(coluna => {

                const cardsColuna = cotacoes.filter(c => c.status === coluna);

                

                const columnEl = document.createElement('div');

                columnEl.className = 'kanban-column';

                columnEl.innerHTML = `

                    <div class="column-header">

                        <span>${coluna}</span>

                        <span class="column-count">${cardsColuna.length}</span>

                    </div>

                    <div class="cards-container" data-coluna="${coluna}" data-tipo="cotacao">

                        ${cardsColuna.length ? cardsColuna.map(card => criarCardCotacao(card)).join('') : 

                            '<div class="empty-state">Nenhum orçamento</div>'}

                    </div>

                `;

                

                container.appendChild(columnEl);

            });

        }

        

        function renderizarKanbanVisitas() {

            const container = document.getElementById('kanbanVisitas');

            container.innerHTML = '';

            

            colunasVisitas.forEach(coluna => {

                const cardsColuna = visitas.filter(v => v.status === coluna);

                

                const columnEl = document.createElement('div');

                columnEl.className = 'kanban-column';

                columnEl.innerHTML = `

                    <div class="column-header">

                        <span>${coluna}</span>

                        <span class="column-count">${cardsColuna.length}</span>

                    </div>

                    <div class="cards-container" data-coluna="${coluna}" data-tipo="visita">

                        ${cardsColuna.length ? cardsColuna.map(card => criarCardVisita(card)).join('') : 

                            '<div class="empty-state">Nenhuma visita</div>'}

                    </div>

                `;

                

                container.appendChild(columnEl);

            });

        }

        

        function criarCardCotacao(card) {

            let borderColor = '#b7ab97';

            if (card.status === 'Pedido Aprovado') borderColor = '#283944';

            if (card.status === 'Pedido Recusado') borderColor = '#dc3545';

            

            return `

                <div class="card" draggable="true" data-id="${card.id}" data-tipo="cotacao" style="border-left-color: ${borderColor}">

                    <div class="card-header">

                        <span class="card-id">${card.id}</span>

                        <span class="badge">${card.dataCriacao}</span>

                    </div>

                    <div class="card-company">${card.empresa || 'Sem empresa'}</div>

                    <div class="card-info">

                        <strong>Contato:</strong> ${card.contato || '---'}<br>

                        <strong>Tel:</strong> ${card.telefone || '---'}<br>

                        <strong>Material:</strong> ${card.material || 'Não especificado'}

                    </div>

                    ${card.arquivos && card.arquivos.length ? 

                        `<div class="card-files">📎 ${card.arquivos.length} arquivo(s)</div>` : ''}

                    <div class="card-actions">

                        <button class="btn-small edit" onclick="editarCard('${card.id}', 'cotacao')">EDITAR</button>

                        <button class="btn-small visita" onclick="criarVisita('${card.id}')">VISITA</button>

                    </div>

                </div>

            `;

        }

        

        function criarCardVisita(card) {

                    </div>

                    <div class="card-actions">

                        <button class="btn-small edit" onclick="editarCard('${card.id}', 'visita')">EDITAR</button>

        function adicionarDragDrop() {

            const cards = document.querySelectorAll('.card[draggable="true"]');

            const containers = document.querySelectorAll('.cards-container');

            

            cards.forEach(card => {

                card.addEventListener('dragstart', handleDragStart);

                card.addEventListener('dragend', handleDragEnd);

            });

            

            containers.forEach(container => {

                container.addEventListener('dragover', handleDragOver);

                container.addEventListener('drop', handleDrop);

            });

        }

        

        let draggedItem = null;

        

        function handleDragStart(e) {

            draggedItem = this;

            this.style.opacity = '0.5';

        }

        

        function handleDragEnd(e) {

            this.style.opacity = '1';

        }

        

        function handleDragOver(e) {

            e.preventDefault();

        }

        

        function handleDrop(e) {

            e.preventDefault();

            

            if (!draggedItem) return;

            

            const container = e.target.closest('.cards-container');

            if (!container) return;

            

            const novaColuna = container.dataset.coluna;

            const tipo = container.dataset.tipo;

            const cardId = draggedItem.dataset.id;

            const cardTipo = draggedItem.dataset.tipo;

            

            if (cardTipo !== tipo) {

                alert('Não é possível mover entre kanbans diferentes');

                return;

            }

            

            if (tipo === 'cotacao') {

                const index = cotacoes.findIndex(c => c.id === cardId);

                if (index !== -1) {

                    cotacoes[index].status = novaColuna;

                    localStorage.setItem('cotacoes', JSON.stringify(cotacoes));

                }

            } else {

                const index = visitas.findIndex(v => v.id === cardId);

                if (index !== -1) {

                    visitas[index].status = novaColuna;

                    localStorage.setItem('visitas', JSON.stringify(visitas));

                }

            }

            

            renderizarKanbans();

        }

        function formatText(type) {

            const textarea = document.querySelector('.rich-textarea');

            if (!textarea) return;

            

            const start = textarea.selectionStart;

            const end = textarea.selectionEnd;

            const text = textarea.value;

            const selectedText = text.substring(start, end);

            

            let formatted = '';

            switch(type) {

                case 'bold': formatted = `<strong>${selectedText}</strong>`; break;

                case 'italic': formatted = `<em>${selectedText}</em>`; break;

                case 'underline': formatted = `<u>${selectedText}</u>`; break;

                case 'h3': formatted = `<h3>${selectedText}</h3>`; break;

            }

            

            textarea.value = text.substring(0, start) + formatted + text.substring(end);

        }
