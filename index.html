    <script>

        const SECRET_CODE = "260378";

        

        const accessCode = document.getElementById('accessCode');

        const codeError = document.getElementById('codeError');

        const formSection = document.getElementById('formSection');

        const fileUploadArea = document.getElementById('fileUploadArea');

        const fileInput = document.getElementById('fileInput');

        const fileList = document.getElementById('fileList');

        const quotationForm = document.getElementById('quotationForm');

        const successMessage = document.getElementById('successMessage');

        

        let uploadedFiles = [];

        

        accessCode.addEventListener('input', function(e) {

            const code = e.target.value;

            

            if (code.length === 6) {

                if (code === SECRET_CODE) {

                    formSection.classList.add('active');

                    codeError.style.display = 'none';

                    accessCode.style.borderColor = '#b7ab97';

                } else {

                    formSection.classList.remove('active');

                    codeError.style.display = 'block';

                    accessCode.style.borderColor = '#dc3545';

                }

            } else {

                formSection.classList.remove('active');

                codeError.style.display = 'none';

                accessCode.style.borderColor = '#ddd';

            }

        });

        

        fileUploadArea.addEventListener('click', () => fileInput.click());

        

        fileUploadArea.addEventListener('dragover', (e) => {

            e.preventDefault();

            fileUploadArea.style.background = '#f8f6f2';

        });

        

        fileUploadArea.addEventListener('dragleave', () => {

            fileUploadArea.style.background = '';

        });

        

        fileUploadArea.addEventListener('drop', (e) => {

            e.preventDefault();

            fileUploadArea.style.background = '';

            handleFiles(e.dataTransfer.files);

        });

        

        fileInput.addEventListener('change', (e) => {

            handleFiles(e.target.files);

        });

        

        function handleFiles(files) {

            for (let file of files) {

                if (file.size <= 10 * 1024 * 1024) {

                    uploadedFiles.push({

                        name: file.name,

                        size: file.size,

                        type: file.type

                    });

                }

            }

            updateFileList();

        }

        

        function updateFileList() {

            if (uploadedFiles.length > 0) {

                fileList.innerHTML = uploadedFiles.map(f => 

                    `<span class="file-item">📄 ${f.name}</span>`

                ).join('');

            } else {

                fileList.innerHTML = '';

            }

        }

        

        quotationForm.addEventListener('submit', function(e) {

            e.preventDefault();

            

            const formData = {

                id: 'COT' + Date.now(),

                empresa: document.getElementById('empresa').value,

                contato: document.getElementById('contato').value,

                email: document.getElementById('email').value,

                telefone: document.getElementById('telefone').value,

                descricao: document.getElementById('descricao').value,

                material: document.getElementById('material').value,

                quantidade: document.getElementById('quantidade').value,

                prazo: document.getElementById('prazo').value,

                infoAdicional: document.getElementById('infoAdicional').value,

                arquivos: uploadedFiles,

                status: 'Novo Orçamento',

                dataCriacao: new Date().toISOString().split('T')[0],

                observacoes: ''

            };

            

            let cotacoes = JSON.parse(localStorage.getItem('cotacoes') || '[]');

            cotacoes.push(formData);

            localStorage.setItem('cotacoes', JSON.stringify(cotacoes));

            

            successMessage.style.display = 'block';

            quotationForm.reset();

            uploadedFiles = [];

            updateFileList();

            

            setTimeout(() => {

                successMessage.style.display = 'none';

            }, 5000);

        });

    </script>
